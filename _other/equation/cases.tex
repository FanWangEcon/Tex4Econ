\section{Two Cases, same line}
\begin{equation*}
f(x) = \begin{cases}
             0  & \text{if } x < 0 \\
             1  & \text{if } x \ge 0
       \end{cases} \quad
g(x) = \begin{cases}
             f(x)+1  & \text{if } x < 0 \\
             f(x)-1  & \text{if } x \ge 0
       \end{cases}
\end{equation*}
